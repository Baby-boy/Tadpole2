log4j.rootLogger = WARN, stdout, file, error

###  print log to console ###
log4j.appender.stdout = org.apache.log4j.ConsoleAppender
log4j.appender.stdout.target = System.out
log4j.appender.stdout.layout = org.apache.log4j.PatternLayout
log4j.appender.stdout.layout.ConversionPattern = %d %p [%c] - %m %n

log4j.logger.com.glanway.iclock = INFO

### write common log to file ###
log4j.appender.file = org.apache.log4j.DailyRollingFileAppender
log4j.appender.file.File = E:/logs/log.log
log4j.appender.file.Append = true
log4j.appender.file.Threshold = INFO 
log4j.appender.file.layout = org.apache.log4j.PatternLayout
log4j.appender.file.layout.ConversionPattern = %d{yyyy-MM-dd HH:mm:ss} - %m%n%n
log4j.appender.file.Encoding=UTF-8

### write error log to file ###
log4j.appender.error = org.apache.log4j.DailyRollingFileAppender
log4j.appender.error.File =  E:/logs/error.log 
log4j.appender.error.Append = true
log4j.appender.error.Threshold = ERROR 
log4j.appender.error.layout = org.apache.log4j.PatternLayout
log4j.appender.error.layout.ConversionPattern = %d{yyyy-MM-dd HH:mm:ss} - %m%n%n
log4j.appender.error.Encoding=UTF-8
